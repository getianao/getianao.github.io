实验一 输出字符串
【实验题目】在屏幕上输出一个字符串
【实验内容】在屏幕上输出字符串：This is my first ASM programme
【实验目的】
1、熟悉汇编语言编程环境
2、熟悉汇编语言上机步骤
3、初步掌握DOS系统功能调用的基本方法

【实验步骤】
1、在DOS提示符下键入EDIT命令，进入DOS的编辑窗口，在该窗口中输入源代码，编辑结束之后一定要以ASM扩展名保存。
2、在DOS提示符下键入MASM命令，对源代码进行编译，若发现有错误，再返回到第一步对源程序的错误之处进行修改，直到不出现错误为止。
3、在DOS提示符下进入LINK命令，产生可执行文件。
4、执行程序，查看实验结果

data segment ;定义数据段
string db 'This is my first ASM program' ,13,10,'$'
;定义了一个字符串,标号是STRING，其值是字符串首字符的地址。
;DB表示的是字符串中每个字符的都是一个字节，每往后加1的时候，地址偏移量加1.
;13 CR 回车
;10 LF 换行
;$作为字符串的结束符
;注意都是单引号
data ends

code segment ;定义代码段
main proc far
    assume cs:code,ds:data
    start:
        push ds
        xor ax,ax
        push ax

        ;把数据段地址装入ds段寄存器
        mov ax,data
        mov ds,ax

        ;将字符串偏移量装进dx数据寄存器
        lea dx,string

        ;输出地址是ds:dx字符串
        mov ah,09h
        int 21h

        ;带返回码结束
        mov ah,4ch
        int 21h
main endp
code ends
    end start

实验二 算术指令综合编程
【实验题目】算术指令综合编程
【实验内容】编程实现：Z((W-X)/5)_Y)2，其中W X Y Z 均为字单元，且为带符号数
【实验目的】
1、熟练掌握算术运算指令
2、初步掌握汇编语言上机调试功能
【实验步骤】
1、源程序的编辑、编译、连接和执行同实验一
2、在DOS命令提示符下键入debug命令（注意：debug 命令后面必须带上可执行的文件名且扩展名不能丢），进入debug环境，在屏幕上出现debug提示符”“
3、键入u反汇编命令，将机器语言反汇编成汇编指令
4、键入g命令设置断点
5、键入t命令单步执行程序
6、键入d命令查看存储单元的内容
7、键入r命令查看寄存器的内容
8、键入q命令退出debug环境，返回DOS

data segment 
w dw ?
x dw ?
y dw ?
z dw ?,?;逗号
data ends

;z=(((w-x)/5)*y)*2
code segment
main proc far
assume cs:code,ds:data
start:
push ds
xor ax,ax
push ax
mov ax,data
mov ds,ax

mov ax,w
mov bx,x
sub ax,bx
cwd;被除数需要符号扩展
mov bx,5
idiv bx;余数放在dx，商放在ax；乘法除法源操作数不能使用立即数寻址方式
mov bx,y
imul bx;和ax相乘，高位放在dx，低位放在ax
;双字乘2的方法
shl ax,1;逻辑左移
rcl dx,1;带进位循环左移
mov z,ax
mov z+2,dx
ret

main endp
code ends
end main;不要忘了这行
