分治法-最近对问题

  • 问题描述
    设p1=(x1, y1), p2=(x2, y2), …, pn=(xn, yn)是平面上n个点构成的集合S，设计算法找出集合S中距离最近的点对。

  • 数据输入
    由文件input.txt提供点的个数及各个点的坐标信息。

    输入文件示例
    *input.txt
    4（数量）
    (2,5)
    (6,3)
    (4,7)
    (1,1)
    (4,7)

  • 结果输出
    程序运行结束时，将距离最近的点对坐标输出到文件output.txt中。

    输出文件示例
    output.txt
    (2,5)

  1. 算法：
    输入：按x坐标升序排列的n个点的集合S={(x1,y1),(x2,y2),…,(xn,yn)}



    2. 输出：最近点对的距离

    递归结束：如果n等于2，则返回（x1,y1）和(x2,y2)之间的距离，算法结束；

    递归过程：
    计算出m=S中各点x坐标的中位数，用这个点把区域划分为两个区域；
    先计算分别在两个区域内的点的最近距离：
    d1=计算={(x1,y1),…,(xm,ym)}的最近对距离
    d2=计算={(xm,ym),…,(xn,yn)}的最近对距离
    d=min{d1,d2};
    下面检查是否存在有跨越两个区域的两个点可以构成最短区域：
    只用考虑距离小于d的两点，这两个点必然存在分界线两端，依次考察集合S中的点p(x,y)，如果x<=xm并且x>=xm-d，则将点p放入集合p1中；如果x>xm并且x<=xm+d，则将点p放入集合p2中；
    对集合p1和p2中的每个点p(x,y)，由鸽舍原理，在y坐标区间[y,y+d]内最多取出6个候选点，计算与点p的最近距离d3；
    返回min{d,d3};

  4. 在oj上做这道题，一直超出内存限制，可能是递归层数太多的原因。后来我在对两个区域的划分方面做了改动，先开始我的做法是找到x坐标中位数的那个点后，递归时，当点的个数为奇数时，左边区域划分为左边界到中位数的点，右边区域划分为中位数的点到右边界，当点的个数为偶数时，左边区域划分为左边界到中位数的较小的点，右边区域划分为中位数较小的点到右边界。我更改后是：递归时，当点的个数为奇数时，左边区域划分为左边界到中位数的点，右边区域划分为中位数的点到右边界，当点的个数为偶数时，左边区域划分为左边界到中位数的较小的点，右边区域划分为中位数较大的点到右边界；就是这么一个我没有注意的瑕疵，让程序的运行效率大大降低。

class ToyPoint
{
public:
    double x, y;//玩具坐标
};

//建立结构体便于迭代函数的结果信息传递
class ReturnPoint
{
public:
    int x, y;//当前最近点对的序号
    double dis;//当前最近点对的距离
    ReturnPoint(int a, int b, double c)//构造函数
    {
        x = a;
        y = b;
        dis = c;
    }
};

bool cmp(ToyPoint a, ToyPoint b)
{
    return a.x < b.x;
}

/**
*求点集中的最近点距离
*@param m 点集区间左端点
*@param n 点集区间右端点
*@return 最短距离
*/
ReturnPoint theClosestDis(int m, int n, ToyPoint *toy)
{
    //递归退出条件
    if (n == m + 1)
    {
        return ReturnPoint(m,n,sqrt(pow((toy[m].x - toy[n].x), 2) + pow((toy[m].y - toy[n].y), 2)));
    }

double d1, d2, d;//储存左右子集的字段最短距离
int midx;//中位数x坐标的点序号

if ((n - m) % 2 == 0)
{
    midx = (m + n) / 2;
    //递归求出左右子集最小距离
    ReturnPoint a = theClosestDis(m, midx, toy);
    ReturnPoint b = theClosestDis(midx, n, toy);
    d1 =a.dis;
    d2 = b.dis;
    d = d1 < d2 ? d1 : d2;//选取两个子集最短距离的较小者
    if (d == d1)
    {
        pointa = a.x;
        pointb = a.y;
    }
    else
    {
        pointa = b.x;
        pointb = b.y;
    }
}
else
{
    midx = (m + n) / 2;
    ReturnPoint a = theClosestDis(m, midx, toy);
    ReturnPoint b = theClosestDis(midx+1, n, toy);
    d1 = a.dis;
    d2 = b.dis;//偶数个点时的划分方法与奇数个时不同，可减少递归次数
    d = d1 < d2 ? d1 : d2;
    if (d == d1)
    {
        pointa = a.x;
        pointb = a.y;
    }
    else
    {
        pointa = b.x;
        pointb = b.y;
    }
}

//提前退出
if (d == 0)
    return ReturnPoint(pointa, pointb, 0);

//求左右子集交接处最小距离
int xm;//中位坐标向左d范围内的最小x坐标点的序号 xm...midx
for (xm = m; xm < midx; xm++)
{
    if (toy[midx].x >= toy[midx].x - d)//进入范围
        break;
}

int xn;//中位坐标向左d范围内的最小x坐标点的序号  midx...mn
for (xn = midx; xn < n; xn++)
{
    if (toy[xn + 1].x >= toy[midx].x + d)//超出范围  
        break;
}

double d3;

for (int i = xm; i <= xn; i++)//对于两边集合里的每个点
{
    int flag = 0;
    for (int j = xm; j <= xn; j++) //在它上方d范围内的点
    {
        if ((toy[j].y - toy[i].y) > d'' (toy[j].y - toy[i].y) < 0)
            continue;
        if (j == i)
            continue;
        if (flag > 6)//flagj>6时右边区间遍历完成
            break;
        flag++;
        d3 = sqrt(pow((toy[i].x - toy[j].x), 2) + pow((toy[i].y - toy[j].y), 2));//检查交界处是否出现最小距离

        if (d3 < d)
        {
            d = d3;
            pointa = i;
            pointb = j;
        }
    }

}

return ReturnPoint(pointa, pointb, d);

}
