线性表-Joseph问题

问题描述：约瑟夫问题的一种描述是：编号为1，2，…，n的n个人按顺时针方向围坐一圈，每人持一个密码（正整数）。一开始任选一个正整数作为报数上限值m，从第一个人开始按顺时针方向自1开始顺序报数，报到m时停止报数。报m的人出列，将他的密码作为新的m值，从他在顺时针方向上的下一个人开始重新从1报数，如此下去，直至所有人全部出列为止。

循环链表方法：

建立循环链表，按照正常过程依次删去链表中死亡的囚犯。
[cpp]

#include
using namespace std;
struct prisoner
{
int number;//囚犯编号
int password;//囚犯密码
prisoner next=NULL;
};
int main()
{
int m;
int amount;//人数
cout<<”输入囚犯人数：”;
cin>>amount;
prisoner list;
prisoner p;
p=&list;
//创建链表
for(int i=0;i<amount;i++)
{ 

    int cinpassword;  
    prisoner *q;  
    q=new prisoner;  
    q->number=i+1;  
    cout<<"当前为"<<i+1<<"号囚犯，请输入囚犯密码:";  
    cin>>cinpassword;  
    q-> number=i+1;  
    q-> password=cinpassword;  
    if(list.next==NULL)  
        {  
        list.next=q;  
        p=q;  
        }             
    else   
    {  
        p->next=q;  
        p=q;  
    }  
}  
p->next=list.next;  
cout<<"请输入初始密码:";  
cin>>m;  
prisoner *node;  
node=&list;  
while(amount>1)  
{  
    for(int j=1;j<m;j++)  
        node=node->next;  
    amount--;  
    cout<<"死亡囚犯的编号为"<<node->next->number<<" 他的密码是"<<node->next->password;  
    cout<<"----剩余人数为:"<<amount<<endl;  
    m=node->next->password;  
    node->next=  node->next->next;  

}   
cout<<"最后存活囚犯编号为"<<node->number<<" 他的密码是"<<node->password<<endl;  
return 0;

}

二位数组方法
[cpp]

#include
using namespace std;
void killPrisoner(int prisoner,int t,int amount)
{
for(int i=t+1;i<amount;i++)
prisoner[i-1]=prisoner[i];//数组依次往前移动 删除数组中一项
}
int main()
{
int prisoner;
int n,password,amount,t=0;
cout<<”请输入囚犯人数；”;
cin>>n;
amount=n;
prisoner=new int*[n];
for(int i=0; i <n; ++i)
prisoner[i] = new int[2];//1存编号，2存密码
for(int i=0;i<n;i++)
{ 

    cout<<"当前为第"<<i+1<<"号囚犯，请输入他的密码：";  
    **(prisoner+i)=i+1;  
    cin>>*(*(prisoner+i)+1);  
}   
cout<<"请输入初始密码；";  
cin>>password;  
for(int i=n;i>1;i--)  
{  
    t=(t+password-1)%amount;//通过用报数后的位置整除以当前人数 算出死亡囚犯在当前队列的位置   
    cout<<"囚犯"<<**(prisoner+t)<<"号已死亡,密码更改为"<< *(*(prisoner+t)+1)<<endl;  
    password=*(*(prisoner+t)+1);  
    killPrisoner(prisoner,t,amount);  
    amount--;  
}  
cout<<"最后存活的囚犯编号为"<< **prisoner;  
return 0;

}
